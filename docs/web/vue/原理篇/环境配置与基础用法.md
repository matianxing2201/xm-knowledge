# 环境配置与基础用法

## Vue 初始化状态流程及对象劫持

### 1、vue2.X版本中，vue 是一个构造函数

vue2中是一个构造函数，如果使用 class 作为入口文件将会非常臃肿，不符合模块化开发的思想。

### 2、函数拓展原型

在 Vue 原型上实现 init 方法，模块化拆分

```javascript
import { initState } from "./observe";

// 使用构造函数 方便扩展直接在原型上进行
function Vue(options) {
  this._init(options);
}

Vue.prototype._init = function (options) {
  let vm = this;
  vm.$options = options; // 将 options 保存在 vm 实例上

  initState(vm); // 初始化状态
};

export default Vue;
```

### 3、初始化状态、响应式数据变化

- 初始化方法 initState -> 初始化数据 initData | initComputed | initWatch
- 由于 `<font style="color:#FA8C16;">data</font>` 有可能是对象，也有可能是函数，所以需要对 `<font style="color:#FA8C16;">data</font>` 进行类型判断，并赋值到 `<font style="color:#FA8C16;">vm._data</font>` 上 `<font style="color:#FA8C16;">data = vm._data = typeofdata === "function" ? data.call(vm) : data || {};</font>`
- 为了避免用户设置与取值的时候需要通过 vm.data ，而是可以直接通过 vm 来设置 data 中的值，所以将 vm.\_data 中的数据做一层代理

```javascript
// 数据代理
function proxy(vm, source, key) {
  Object.defineProperty(vm, key, {
    get() {
      return vm[source][key]; // vm.message -> vm._data.message
    },
    set(newValue) {
      vm[source][key] = newValue;
    },
  });
}
```

- 通过 observe 方法将对象进行劫持（Object.defineProperty）

```javascript
export function defineReactive(data, key, value) {
  observer(value); //递归
  // 不兼容IE8 及以下
  Object.defineProperty(data, key, {
    get() {
      console.log("获取了value");
      return value;
    },
    set(newValue) {
      if (newValue === value) return;
      console.log("修改了value：" + newValue);
      value = newValue;
    },
  });
}
class Observe {
  constructor(data) {
    this.walk(data); //  将对象劫持
  }

  walk(data) {
    // 使用 for in 循环会遍历原型上的属性。 通过 Object.keys 将对象的key 作为数组 便利
    let keys = Object.keys(data);
    for (let i = 0; i < keys.length; i++) {
      let key = keys[i],
        value = data[key];
      defineReactive(data, key, value);
    }
  }
}

export default Observe;
```

## 数组劫持

通过原型拦截的方式重写了数组的7个方法，首先获取数组的 `<font style="color:#FA8C16;">Observer</font>`。如果有新值，调用 `<font style="color:#FA8C16;">observeArray</font>` 对新的值进行监听，然后调用 notify，通知 reander watcher，执行update。

arrayMethods 首先继承 Array，然后对数组中所有能改变数组自身得方法，如 push、pop等这些方法进行重写。重写后的方法先执行他们本身原有的逻辑，并对能增加数组长度的方法（push、unshift、splice）方法做判断，获取插入的值，然后把新添加的值变成一个 <font style="color:#FA8C16;">响应式对象 </font>，并在调用 `<font style="color:#FA8C16;">ob.dep.notify</font>` 手动触发通知依赖。将当前数组的变化通知给订阅者。

```javascript
const arrayProto = Array.prototype; //原生Array的原型
export const arrayMethods = Object.create(arrayProto);
["push", "pop", "shift", "unshift", "splice", "sort", "reverse"].forEach(
  function (method) {
    const original = arrayProto[method]; //缓存元素数组原型
    //这里重写了数组的几个原型方法
    def(arrayMethods, method, function mutator() {
      //这里备份一份参数应该是从性能方面的考虑
      let i = arguments.length;
      const args = new Array(i);
      while (i--) {
        args[i] = arguments[i];
      }
      const result = original.apply(this, args); //原始方法求值
      const ob = this.__ob__; //这里this.__ob__指向的是数据的Observer
      let inserted;
      switch (method) {
        case "push":
          inserted = args;
          break;
        case "unshift":
          inserted = args;
          break;
        case "splice":
          inserted = args.slice(2);
          break;
      }
      if (inserted) ob.observeArray(inserted);
      // notify change
      ob.dep.notify();
      return result;
    });
  },
);

//定义属性
function def(obj, key, val, enumerable) {
  Object.defineProperty(obj, key, {
    value: val,
    enumerable: !!enumerable,
    writable: true,
    configurable: true,
  });
}
```
